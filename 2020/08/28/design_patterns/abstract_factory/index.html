    Abstract Factory模式(抽象工厂模式)

    设计模式 设计模式
    发布时间 : 2020-08-28 01:45
    阅读 :
    1. 什么是抽象工厂模式？
    2. 示范代码(golang)

    什么是抽象工厂模式？

    • 抽象工厂模式即将关联零件组装成产品
    • 将抽象零件组装成为抽象产品

    示范代码(golang)

    新建一个零食工厂

    package Abstract_Factory

import "fmt"

// 创建一个工厂，生成饮料、零食

// 抽象产品
type Product interface {
    Name() string
}

func NewDrink(name string) *Drink {
    return &Drink{name: name}
}

// 具体产品 饮料
type Drink struct {
    name string
}

func (d *Drink) Name() string {
    return d.name
}

func NewSnack(name string) *Snack {
    return &Snack{name: name}
}

// 具体产品 零食
type Snack struct {
    name string
}

func (s *Snack) Name() string {
    return s.name
}

// 抽象工厂
type Factory interface {
    CreateDrink(name string)
    CreateSnack(name string)
    AddProduct(product Product)
    GetProduct() []Product
}

func NewSimpleFactory() *SimpleFactory {
    fmt.Println("新建一个零食工产")
    return &SimpleFactory{}
}

// 具体工厂 零食工厂
type SimpleFactory struct {
    products []Product
}

func (f *SimpleFactory) CreateDrink(name string) {
    d := NewDrink(name)
    fmt.Printf("生产一瓶%s\n", d.name)
    f.AddProduct(d)
    return
}

func (f *SimpleFactory) CreateSnack(name string) {
    s := NewSnack(name)
    fmt.Printf("生产一包%s\n", s.name)
    f.AddProduct(s)
    return
}
func (f *SimpleFactory) AddProduct(product Product) {
    f.products = append(f.products, product)
    return
}

func (f *SimpleFactory) GetProduct() []Product {
    fmt.Println("现有产品：")
    for index, p := range f.products {
        fmt.Println(index+1, p.Name())
    }
    return f.products
}
    package Abstract_Factory

import (
    "testing"
)

func TestSimpleFactory(t *testing.T) {
    var factory Factory = NewSimpleFactory()
    factory.CreateDrink("苹果醋")
    factory.CreateSnack("小熊饼干")
    factory.GetProduct()
}
